+ /

- /proc/**
- /sys/**
- /tmp/**
- /run/**
- /mnt/**
- /media/**

- /var/tmp/**

- /home/*/.cache/**
- /home/*/Dropbox/**
- /home/*/.dropbox/**
- /home/*/tmp/**
